def ensure_columns(df):
    data = df.copy()
    defaults = {c: None for c in COLUMNS}
    defaults.update({"경마장":"", "경주마":"", "줄":"", "기수":"", "성별":"", "부마":"", "모마":""})
    for c, v in defaults.items():
        if c not in data.columns:
            data[c] = v
    data["경주일"] = pd.to_datetime(data["경주일"], errors="coerce")
    for c in [x for x in COLUMNS if x not in ["경주일","경마장","경주마","줄","기수","성별","부마","모마"]]:
        data[c] = pd.to_numeric(data[c], errors="coerce")
    data["줄"] = data.apply(lambda r: r["줄"] if str(r["줄"]).strip() not in ["", "None", "nan"] else (f"{int(r['마번'])}째줄" if pd.notna(r["마번"]) else ""), axis=1)
    return data

def normalize(s, high_is_good=True, default=55):
    s = pd.to_numeric(s, errors="coerce")
    if s.notna().sum() == 0:
        return pd.Series([default]*len(s), index=s.index)
    mn, mx = s.min(), s.max()
    if mn == mx:
        return pd.Series([default]*len(s), index=s.index)
    val = (s-mn)/(mx-mn)*100
    if not high_is_good:
        val = 100-val
    return val.fillna(default)

def clamp(s):
    return pd.to_numeric(s, errors="coerce").fillna(55).clip(0,100)

def age_engine(age):
    a = pd.to_numeric(age, errors="coerce")
    return a.apply(lambda x: 75 if pd.isna(x) else max(25, min(100, 100-abs(x-4.5)*18)))

def rest_engine(days):
    d = pd.to_numeric(days, errors="coerce")
    return d.apply(lambda x: 60 if pd.isna(x) else max(30, 100-abs(x-21)*2.2))

def body_engine(change):
    c = pd.to_numeric(change, errors="coerce")
    return c.apply(lambda x: 60 if pd.isna(x) else max(30, 100-abs(x)*8))

def improvement_score(row):
    prev2, prev1, now = row.get("지지난순위"), row.get("직전순위"), row.get("순위")
    score = 50
    if pd.notna(prev2) and pd.notna(prev1):
        score += max(-20, min(25, (prev2-prev1)*8))
    if pd.notna(prev1) and pd.notna(now):
        score += max(-25, min(35, (prev1-now)*10))
    return max(0, min(100, score))

def change_reason(row):
    r=[]
    if pd.notna(row.get("지지난순위")) and pd.notna(row.get("직전순위")) and row["직전순위"] < row["지지난순위"]:
        r.append(f"지지난 {int(row['지지난순위'])}위→직전 {int(row['직전순위'])}위 상승")
    if pd.notna(row.get("직전순위")) and pd.notna(row.get("순위")) and row["순위"] < row["직전순위"]:
        r.append(f"직전 {int(row['직전순위'])}위→이번 {int(row['순위'])}위 상승")
    if pd.notna(row.get("체중증감")) and abs(row["체중증감"]) <= 2: r.append("체중 안정")
    if pd.notna(row.get("출전간격일")) and 14 <= row["출전간격일"] <= 28: r.append("출전간격 안정")
    if pd.notna(row.get("컨디션점수")) and row["컨디션점수"] >= 80: r.append("컨디션 양호")
    if pd.notna(row.get("트랙적성")) and row["트랙적성"] >= 80: r.append("트랙 적성")
    return " / ".join(r) if r else "종합 점수 기준"

def reason(row):
    r=[]
    for name, col, th in [("기수","기수엔진",75),("혈통","혈통엔진",80),("최근성적","최근성적엔진",75),("스피드","스피드엔진",82),("지구력","지구력엔진",82),("트랙","트랙엔진",82),("거리","거리엔진",82),("컨디션","컨디션엔진",82),("상승세","상승세엔진",75)]:
        if row.get(col,0) >= th: r.append(name)
    return " / ".join(r) if r else "종합균형"

def run_engine(df, weights):
    data = ensure_columns(df)
    data["기수엔진"] = normalize(data["기수승률"], True, 55)*.55 + normalize(data["기수복승률"], True, 55)*.45
    data["혈통엔진"] = clamp(data["혈통점수"])
    data["나이엔진"] = age_engine(data["나이"])
    data["최근성적엔진"] = normalize(data["최근3경주평균순위"].fillna(data["순위"]), False, 55)*.45 + normalize(data["최근5경주평균순위"].fillna(data["순위"]), False, 55)*.25 + normalize(data["최근승률"], True, 55)*.30
    data["스피드엔진"] = clamp(data["스피드점수"])*.65 + clamp(data["순발력점수"])*.35
    endurance = clamp(data["지구력점수"])*.55 + clamp(data["막판추입점수"])*.45
    data["지구력엔진"] = (endurance * data["거리"].apply(lambda x: 1.08 if pd.notna(x) and x >= 1600 else 1.0)).clip(upper=100)
    data["트랙엔진"] = clamp(data["트랙적성"])*.65 + clamp(data["주로상태적성"])*.35
    data["거리엔진"] = clamp(data["거리적성"])
    data["부담중량엔진"] = normalize(data["부담중량"], False, 55)
    data["컨디션엔진"] = clamp(data["컨디션점수"])*.45 + clamp(data["훈련점수"])*.35 + rest_engine(data["출전간격일"])*.20
    data["체중변화엔진"] = body_engine(data["체중증감"])
    data["배당안정성엔진"] = normalize(data["배당률"], False, 55)
    data["상승세엔진"] = data.apply(improvement_score, axis=1)
    data["한달변경요인"] = data.apply(change_reason, axis=1)
    data["리스크방어점수"] = (data["컨디션엔진"]*.25 + data["체중변화엔진"]*.20 + data["부담중량엔진"]*.18 + data["배당안정성엔진"]*.12 + data["최근성적엔진"]*.15 + data["상승세엔진"]*.10).round(2)
    data["코스전략엔진"] = data.apply(lambda r: (r["스피드엔진"]*.55+r["기수엔진"]*.20+r["거리엔진"]*.25) if pd.notna(r["거리"]) and r["거리"]<=1200 else (r["지구력엔진"]*.45+r["거리엔진"]*.30+r["트랙엔진"]*.25) if pd.notna(r["거리"]) and r["거리"]>=1600 else (r["스피드엔진"]*.30+r["지구력엔진"]*.30+r["거리엔진"]*.25+r["기수엔진"]*.15), axis=1).round(2)
    data["엔진종합점수"] = sum(data[k]*v for k,v in weights.items()).round(2)
    data["최종추천점수"] = (data["엔진종합점수"]*.80 + data["리스크방어점수"]*.20).round(2)
    data["추천등급"] = data.apply(lambda r: "강추천" if r["최종추천점수"]>=78 and r["리스크방어점수"]>=65 else "추천" if r["최종추천점수"]>=70 else "관찰" if r["최종추천점수"]>=62 else "제외", axis=1)
    data["리스크등급"] = data.apply(lambda r: "A 안정권" if r["최종추천점수"]>=78 and r["리스크방어점수"]>=65 else "B 관심권" if r["최종추천점수"]>=68 else "C 보류권" if r["최종추천점수"]>=58 else "D 위험권", axis=1)
    data["선정이유"] = data.apply(reason, axis=1)
    return data.sort_values("최종추천점수", ascending=False)

@st.cache_data(ttl=45, show_spinner=False)
def fetch_html_tables(url):
    headers = {"User-Agent":"Mozilla/5.0", "Accept-Language":"ko-KR,ko;q=0.9,en;q=0.8"}
    try:
        r = requests.get(url, headers=headers, timeout=12)
        r.raise_for_status()
        tables = pd.read_html(r.text)
        clean = []
        for t in tables:
            t = t.dropna(how="all").dropna(axis=1, how="all")
            if len(t) > 0:
                t.columns = [str(c).replace("\n"," ").strip() for c in t.columns]
                clean.append(t)
        return {"ok": True, "url": url, "tables": clean, "count": len(clean), "error": ""}
    except Exception as e:
        return {"ok": False, "url": url, "tables": [], "count": 0, "error": str(e)}

def pick_col(df, terms):
    for term in terms:
        for c in df.columns:
            if term in str(c):
                return c
    return None

            if res["ok"]:
                for t in res["tables"]:
                    n = normalize_kra_table(course, t)
                    if len(n) and "경주마" in n.columns:
                        pieces.append(n)
    live_df = pd.concat(pieces, ignore_index=True).drop_duplicates() if pieces else pd.DataFrame()
    return result, live_df

def detect_changes(prev, curr):
    prev = ensure_columns(prev) if len(prev) else pd.DataFrame()
    curr = ensure_columns(curr) if len(curr) else pd.DataFrame()
    if prev.empty or curr.empty:
        return pd.DataFrame(columns=["구분","경마장","경주마","항목","이전","현재","상태"])
    key_cols = ["경마장","경주번호","경주마"]
    changes = []
    p = prev.set_index(key_cols, drop=False)
    c = curr.set_index(key_cols, drop=False)
    for key, row in c.iterrows():
        if key not in p.index:
            changes.append(["신규", row.get("경마장",""), row.get("경주마",""), "출전", "-", "신규", "변경"])
            continue
        old = p.loc[key]
        if isinstance(old, pd.DataFrame): old = old.iloc[0]
        for col, label in [("기수","기수"),("마번","마번"),("줄","줄"),("거리","트랙/거리"),("부담중량","부담중량"),("배당률","배당률"),("순위","순위")]:
            ov, nv = old.get(col), row.get(col)
            if str(ov) != str(nv) and not (pd.isna(ov) and pd.isna(nv)):
                changes.append(["변경", row.get("경마장",""), row.get("경주마",""), label, ov, nv, "즉시 재분석"])
    for key, row in p.iterrows():
        if key not in c.index:
            changes.append(["삭제", row.get("경마장",""), row.get("경주마",""), "출전", "있음", "삭제/누락", "확인"])
    return pd.DataFrame(changes, columns=["구분","경마장","경주마","항목","이전","현재","상태"])

def monthly_analysis(engine_df, days=31):
    max_date = engine_df["경주일"].max()
    if pd.isna(max_date): max_date = pd.Timestamp.today()
    start = max_date - pd.Timedelta(days=days)
    month = engine_df[(engine_df["경주일"] >= start) & (engine_df["경주일"] <= max_date)].copy()
    winners = month[month["순위"] == 1].copy().sort_values("경주일", ascending=False)
    pattern = {}
    if len(winners):
        pattern = {
            "우승마수": len(winners),
            "평균지지난순위": winners["지지난순위"].mean(),
            "평균직전순위": winners["직전순위"].mean(),
            "평균트랙적성": winners["트랙적성"].mean(),
            "평균컨디션": winners["컨디션점수"].mean(),
            "평균스피드": winners["스피드점수"].mean(),
            "평균지구력": winners["지구력점수"].mean(),
            "평균출전간격": winners["출전간격일"].mean(),
        }
    return month, winners, pattern, start, max_date

def clean_phone(p):
    return "".join(ch for ch in str(p) if ch.isdigit())

def send_sms_solapi(api_key, api_secret, from_phone, to_phone, text):
    try:
        from solapi import SolapiMessageService
        from solapi.model import RequestMessage
    except Exception:
        return False, "solapi 패키지가 없습니다. requirements.txt에 포함되어 있지만 설치가 실패했을 수 있습니다."
    try:
        service = SolapiMessageService(api_key=api_key, api_secret=api_secret)
        msg = RequestMessage(from_=from_phone, to=to_phone, text=text)
        res = service.send(msg)
        return True, f"문자 발송 요청 완료: {res}"
    except Exception as e:
        return False, f"문자 발송 실패: {e}"
